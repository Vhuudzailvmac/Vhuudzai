--// GLOBAL FLAGS
getgenv().AutoQuest = false
getgenv().AutoBoss = false
getgenv().AutoRaid = false
getgenv().AutoFruit = false
getgenv().FastAttack = true

--// QUEST DATA
local function GetQuestData()
    local Lv = LocalPlayer.Data.Level.Value

    if Sea1 then
        if Lv <= 9 then
            return "BanditQuest1",1,"Bandit"
        elseif Lv <= 29 then
            return "JungleQuest",2,"Gorilla"
        elseif Lv <= 59 then
            return "BuggyQuest1",1,"Pirate"
        elseif Lv <= 99 then
            return "DesertQuest",1,"Desert Bandit"
        end
    elseif Sea2 then
        if Lv <= 724 then
            return "Area1Quest",1,"Raider"
        elseif Lv <= 774 then
            return "Area1Quest",2,"Mercenary"
        elseif Lv <= 899 then
            return "Area2Quest",1,"Swan Pirate"
        end
    elseif Sea3 then
        if Lv <= 1574 then
            return "PiratePortQuest",1,"Pirate Millionaire"
        elseif Lv <= 1724 then
            return "AmazonQuest",1,"Dragon Crew Warrior"
        end
    end
end

--// AUTO QUEST
task.spawn(function()
    while task.wait(0.3) do
        if getgenv().AutoQuest then
            local quest, level, mob = GetQuestData()
            if quest then
                pcall(function()
                    CommF:InvokeServer("StartQuest", quest, level)
                end)
                for _,v in pairs(Workspace.Enemies:GetChildren()) do
                    if v.Name == mob and v:FindFirstChild("HumanoidRootPart") and v.Humanoid.Health > 0 then
                        LocalPlayer.Character.HumanoidRootPart.CFrame =
                            v.HumanoidRootPart.CFrame * CFrame.new(0,0,4)
                        break
                    end
                end
            end
        end
    end
end)

--// AUTO BOSS
task.spawn(function()
    while task.wait(0.5) do
        if getgenv().AutoBoss then
            for _,boss in pairs(Workspace.Enemies:GetChildren()) do
                if boss:FindFirstChild("Humanoid")
                and boss.Humanoid.Health > 0
                and boss.Name:lower():find("boss") then
                    LocalPlayer.Character.HumanoidRootPart.CFrame =
                        boss.HumanoidRootPart.CFrame * CFrame.new(0,0,6)
                end
            end
        end
    end
end)

--// AUTO RAID
task.spawn(function()
    while task.wait(1) do
        if getgenv().AutoRaid then
            pcall(function()
                CommF:InvokeServer("RaidsNpc","Select","Flame")
                CommF:InvokeServer("RaidsNpc","Start")
            end)
        end
    end
end)

--// AUTO FRUIT
task.spawn(function()
    while task.wait(1) do
        if getgenv().AutoFruit then
            for _,v in pairs(Workspace:GetChildren()) do
                if v:IsA("Tool") and v:FindFirstChild("Handle") then
                    LocalPlayer.Character.HumanoidRootPart.CFrame = v.Handle.CFrame
                end
            end
        end
    end
end)

--// FAST ATTACK (GRAVITY)
task.spawn(function()
    local Net = require(ReplicatedStorage.Modules.Net)
    while task.wait(0.05) do
        if getgenv().FastAttack then
            pcall(function()
                Net["RE/RegisterAttack"]:FireServer()
                Net["RE/RegisterHit"]:FireServer()
            end)
        end
    end
end)
