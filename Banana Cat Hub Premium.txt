local function CreateToggle(text, y)
	local frame = Instance.new("Frame", content)
	frame.Size = UDim2.new(1,0,0,45)
	frame.Position = UDim2.fromOffset(0,y)
	frame.BackgroundColor3 = Color3.fromRGB(30,30,30)
	Instance.new("UICorner", frame).CornerRadius = UDim.new(0,10)

	local lbl = Instance.new("TextLabel", frame)
	lbl.Text = text
	lbl.Size = UDim2.new(1,-60,1,0)
	lbl.Position = UDim2.fromOffset(15,0)
	lbl.TextXAlignment = Left
	lbl.TextColor3 = Color3.fromRGB(230,230,230)
	lbl.Font = Enum.Font.Gotham
	lbl.TextSize = 14
	lbl.BackgroundTransparency = 1

	local btn = Instance.new("TextButton", frame)
	btn.Size = UDim2.fromOffset(26,26)
	btn.Position = UDim2.new(1,-36,0.5,-13)
	btn.Text = ""
	btn.BackgroundColor3 = Color3.fromRGB(40,40,40)
	Instance.new("UICorner", btn).CornerRadius = UDim.new(1,0)

	local on = false
	btn.MouseButton1Click:Connect(function()
		on = not on
		TweenService:Create(
			btn,
			TweenInfo.new(0.2),
			{BackgroundColor3 = on and Color3.fromRGB(0,255,200) or Color3.fromRGB(40,40,40)}
		):Play()
	end)
end

local function toggleItem(text, y)
    local f = Instance.new("Frame", content)
    f.Size = UDim2.new(1,0,0,42)
    f.Position = UDim2.new(0,0,0,y)
    f.BackgroundColor3 = Color3.fromRGB(32,32,32)
    f.BorderSizePixel = 0
    Instance.new("UICorner", f).CornerRadius = UDim.new(0,6)

    local lbl = Instance.new("TextLabel", f)
    lbl.Size = UDim2.new(1,-60,1,0)
    lbl.Position = UDim2.new(0,12,0,0)
    lbl.Text = text
    lbl.Font = Enum.Font.Gotham
    lbl.TextSize = 14
    lbl.TextXAlignment = Left
    lbl.TextColor3 = Color3.fromRGB(230,230,230)
    lbl.BackgroundTransparency = 1

    local box = Instance.new("Frame", f)
    box.Size = UDim2.new(0,22,0,22)
    box.Position = UDim2.new(1,-34,0.5,-11)
    box.BackgroundColor3 = Color3.fromRGB(20,20,20)
    box.BorderSizePixel = 0
    Instance.new("UICorner", box).CornerRadius = UDim.new(0,4)

    local outline = Instance.new("UIStroke", box)
    outline.Color = Color3.fromRGB(255,215,120)

    local on = false
    box.InputBegan:Connect(function()
        on = not on
        TweenService:Create(box, TweenInfo.new(0.15), {
            BackgroundColor3 = on and Color3.fromRGB(255,215,120) or Color3.fromRGB(20,20,20)
        }):Play()
    end)
end
