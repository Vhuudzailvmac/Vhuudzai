local pages = {}

local function newPage()
	local p = Instance.new("ScrollingFrame", main)
	p.Position = UDim2.new(0,170,0,50)
	p.Size = UDim2.new(1,-170,1,-50)
	p.ScrollBarImageTransparency = 1
	p.CanvasSize = UDim2.new(0,0,0,0)
	p.BackgroundTransparency = 1
	p.Visible = false
	table.insert(pages, p)
	return p
end

local function tabBtn(text, page, y)
	local b = Instance.new("TextButton", tabs)
	b.Position = UDim2.new(0,10,0,y)
	b.Size = UDim2.new(1,-20,0,40)
	b.Text = text
	b.Font = Enum.Font.Gotham
	b.TextSize = 14
	b.TextColor3 = Color3.new(1,1,1)
	b.BackgroundColor3 = Color3.fromRGB(35,35,35)
	Instance.new("UICorner", b)

	b.MouseButton1Click:Connect(function()
		for _,p in pairs(pages) do p.Visible = false end
		page.Visible = true
	end)
end

local function toggle(parent, text, y)
	local btn = Instance.new("TextButton", parent)
	btn.Position = UDim2.new(0,20,0,y)
	btn.Size = UDim2.new(0,300,0,42)
	btn.Text = "[ OFF ] "..text
	btn.Font = Enum.Font.Gotham
	btn.TextSize = 14
	btn.TextColor3 = Color3.new(1,1,1)
	btn.BackgroundColor3 = Color3.fromRGB(40,40,40)
	Instance.new("UICorner", btn)

	local on = false
	btn.MouseButton1Click:Connect(function()
		on = not on
		btn.Text = (on and "[ ON ] " or "[ OFF ] ")..text
		btn.BackgroundColor3 = on and COLOR_YELLOW or Color3.fromRGB(40,40,40)
		btn.TextColor3 = on and Color3.fromRGB(0,0,0) or Color3.new(1,1,1)
	end)

	parent.CanvasSize = UDim2.new(0,0,0,y+70)
end
