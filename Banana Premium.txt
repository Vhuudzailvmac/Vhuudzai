local pages = {}

local function newPage()
	local page = Instance.new("ScrollingFrame", content)
	page.Size = UDim2.new(1,0,1,0)
	page.CanvasSize = UDim2.new(0,0,0,800)
	page.ScrollBarImageTransparency = 0.3
	page.BackgroundTransparency = 1

	Instance.new("UIListLayout", page).Padding = UDim.new(0,12)
	return page
end

-- TAB BUTTON
local function tab(text, page, y)
	local b = Instance.new("TextButton", menu)
	b.Size = UDim2.new(1,-20,0,40)
	b.Position = UDim2.new(0,10,0,y)
	b.Text = text
	b.Font = Enum.Font.GothamBold
	b.TextSize = 14
	b.TextColor3 = Color3.new(1,1,1)
	b.BackgroundColor3 = Color3.fromRGB(35,35,35)
	Instance.new("UICorner", b)

	b.MouseButton1Click:Connect(function()
		for _,p in pairs(pages) do p.Visible = false end
		page.Visible = true
	end)
end

-- TOGGLE (PREMIUM STYLE)
local function toggle(parent, text)
	local box = Instance.new("Frame", parent)
	box.Size = UDim2.new(1,-40,0,46)
	box.BackgroundColor3 = Color3.fromRGB(28,28,28)
	Instance.new("UICorner", box)

	local lb = Instance.new("TextLabel", box)
	lb.Size = UDim2.new(1,-80,1,0)
	lb.Position = UDim2.new(0,12,0,0)
	lb.Text = text
	lb.Font = Enum.Font.Gotham
	lb.TextSize = 14
	lb.TextXAlignment = Left
	lb.TextColor3 = Color3.new(1,1,1)
	lb.BackgroundTransparency = 1

	local sw = Instance.new("Frame", box)
	sw.Size = UDim2.new(0,46,0,22)
	sw.Position = UDim2.new(1,-58,0.5,-11)
	sw.BackgroundColor3 = Color3.fromRGB(90,90,90)
	Instance.new("UICorner", sw).CornerRadius = UDim.new(1,0)

	local k = Instance.new("Frame", sw)
	k.Size = UDim2.new(0,18,0,18)
	k.Position = UDim2.new(0,2,0.5,-9)
	k.BackgroundColor3 = Color3.fromRGB(230,230,230)
	Instance.new("UICorner", k).CornerRadius = UDim.new(1,0)

	local on = false
	sw.InputBegan:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 then
			on = not on
			if on then
				sw.BackgroundColor3 = Color3.fromRGB(0,170,0)
				k:TweenPosition(UDim2.new(1,-20,0.5,-9),"Out","Quad",0.15,true)
			else
				sw.BackgroundColor3 = Color3.fromRGB(90,90,90)
				k:TweenPosition(UDim2.new(0,2,0.5,-9),"Out","Quad",0.15,true)
			end
		end
	end)
end
