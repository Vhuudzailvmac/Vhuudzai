-- PAGE HOLDER
local Pages = {}

-- CREATE PAGE
local function CreatePage(name)
	local page = Instance.new("ScrollingFrame", Content)
	page.Size = UDim2.new(1,0,1,0)
	page.CanvasSize = UDim2.new(0,0,0,0)
	page.ScrollBarImageTransparency = 1
	page.Visible = false
	page.BackgroundTransparency = 1

	local list = Instance.new("UIListLayout", page)
	list.Padding = UDim.new(0,10)

	local pad = Instance.new("UIPadding", page)
	pad.PaddingTop = UDim.new(0,12)
	pad.PaddingLeft = UDim.new(0,12)
	pad.PaddingRight = UDim.new(0,12)

	Pages[name] = page
	return page, list
end

-- CREATE TAB BUTTON
local CurrentTab
local function CreateTab(name)
	local btn = Instance.new("TextButton", Side)
	btn.Size = UDim2.new(1,-16,0,38)
	btn.Text = name
	btn.Font = Enum.Font.Gotham
	btn.TextSize = 14
	btn.TextColor3 = C_TEXT
	btn.BackgroundColor3 = Color3.fromRGB(32,32,32)
	Instance.new("UICorner", btn).CornerRadius = UDim.new(0,8)

	btn.MouseButton1Click:Connect(function()
		if CurrentTab then
			CurrentTab.BackgroundColor3 = Color3.fromRGB(32,32,32)
		end
		CurrentTab = btn
		btn.BackgroundColor3 = C_YELLOW

		for _,p in pairs(Pages) do
			p.Visible = false
		end
		Pages[name].Visible = true
	end)

	return btn
end

-- CREATE TOGGLE (BANANA STYLE)
local function CreateToggle(parent, text)
	local item = Instance.new("Frame", parent)
	item.Size = UDim2.new(1,-10,0,42)
	item.BackgroundColor3 = C_ITEM
	Instance.new("UICorner", item).CornerRadius = UDim.new(0,10)

	local label = Instance.new("TextLabel", item)
	label.Size = UDim2.new(1,-80,1,0)
	label.Position = UDim2.new(0,12,0,0)
	label.Text = text
	label.Font = Enum.Font.Gotham
	label.TextSize = 14
	label.TextXAlignment = Left
	label.TextColor3 = C_TEXT
	label.BackgroundTransparency = 1

	local toggle = Instance.new("Frame", item)
	toggle.Size = UDim2.new(0,44,0,22)
	toggle.Position = UDim2.new(1,-60,0.5,-11)
	toggle.BackgroundColor3 = Color3.fromRGB(70,70,70)
	Instance.new("UICorner", toggle).CornerRadius = UDim.new(1,0)

	local dot = Instance.new("Frame", toggle)
	dot.Size = UDim2.new(0,18,0,18)
	dot.Position = UDim2.new(0,2,0.5,-9)
	dot.BackgroundColor3 = Color3.fromRGB(220,220,220)
	Instance.new("UICorner", dot).CornerRadius = UDim.new(1,0)

	local on = false
	item.InputBegan:Connect(function()
		on = not on
		if on then
			toggle.BackgroundColor3 = C_YELLOW
			dot:TweenPosition(UDim2.new(1,-20,0.5,-9),"Out","Quad",0.15,true)
		else
			toggle.BackgroundColor3 = Color3.fromRGB(70,70,70)
			dot:TweenPosition(UDim2.new(0,2,0.5,-9),"Out","Quad",0.15,true)
		end
	end)
end
